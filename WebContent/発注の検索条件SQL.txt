select 商品マスタ.商品ID,商品名,カテゴリ名,仕入基準単価,販売単価,安全在庫数,COALESCE(sum(在庫数),0) as 在庫数
from 商品マスタ inner join カテゴリマスタ
on 商品マスタ.カテゴリID = カテゴリマスタ.カテゴリID
left outer join 在庫
on 商品マスタ.商品ID = 在庫.商品ID
where 商品名 like '%の%'
and 商品マスタ.カテゴリID = '01'
group by 商品マスタ.商品ID,商品名,カテゴリ名,仕入基準単価,販売単価,安全在庫数
order by 商品マスタ.商品ID
